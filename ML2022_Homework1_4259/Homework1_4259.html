#The probability density function for the Gaussian/Normal distribution
def normal_distribution_PDF(x,μ,σ_square):
    return (1/(sqrt(2 * pi * σ_square))) * exp((- (x - μ)**2) / (2 * σ_square))
    
#Calculate the conditional probability of the test record given a type
def calculate_conditional_probability(type_,bl,rf,hl,hs,c):
    if type_  == "Ghoul":
        return pr_ghoul*normal_distribution_PDF(bl,ghoul_mean_bl,ghoul_var_bl)*normal_distribution_PDF(rf,ghoul_mean_rf,ghoul_var_rf)*normal_distribution_PDF(hl,ghoul_mean_hl,ghoul_var_hl)*normal_distribution_PDF(hs,ghoul_mean_hs,ghoul_var_hs)*normal_distribution_PDF(c,ghoul_mean_c,ghoul_var_c)
    elif type_ == "Goblin":
        return pr_goblin*normal_distribution_PDF(bl,goblin_mean_bl,goblin_var_bl)*normal_distribution_PDF(rf,goblin_mean_rf,goblin_var_rf)*normal_distribution_PDF(hl,goblin_mean_hl,goblin_var_hl)*normal_distribution_PDF(hs,goblin_mean_hs,goblin_var_hs)*normal_distribution_PDF(c,goblin_mean_c,goblin_var_c)
    else:
        return pr_ghost*normal_distribution_PDF(bl,ghost_mean_bl,ghost_var_bl)*normal_distribution_PDF(rf,ghost_mean_rf,ghost_var_rf)*normal_distribution_PDF(hl,ghost_mean_hl,ghost_var_hl)*normal_distribution_PDF(hs,ghost_mean_hs,ghost_var_hs)*normal_distribution_PDF(c,ghost_mean_c,ghost_var_c)

#for every attribute of every test record calculate the pdf for every type
for i in range(len(test_df)):

    p_type_is_ghoul_for_given_color          = normal_distribution_PDF(color,ghoul_mean_c,ghoul_var_c)
    
    p_type_is_goblin_for_given_bone_length   = normal_distribution_PDF(bone_length,goblin_mean_bl,goblin_var_bl)
    p_type_is_goblin_for_given_rotting_flesh = normal_distribution_PDF(rotting_flesh,goblin_mean_rf,goblin_var_rf)
    p_type_is_goblin_for_given_hair_length   = normal_distribution_PDF(hair_length,goblin_mean_hl,goblin_var_hl)

    p_type_is_goblin_for_given_color         = normal_distribution_PDF(color,goblin_mean_c,goblin_var_c)
    
    p_type_is_ghost_for_given_bone_length    = normal_distribution_PDF(bone_length,ghost_mean_bl,ghost_var_bl)
